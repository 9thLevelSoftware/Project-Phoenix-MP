**Add ensureValidToken():**

```kotlin
/**
 * Ensures the access token is valid before making an authenticated request.
 * If expired, attempts a single refresh (serialized by Mutex).
 * Returns the valid token, or null if refresh failed (auth cleared).
 */
private suspend fun ensureValidToken(): String? {
    val currentToken = tokenStorage.getToken()
    if (currentToken == null) return null

    if (!tokenStorage.isTokenExpired()) return currentToken

    // Token expired — attempt refresh (serialized)
    return refreshMutex.withLock {
        // Double-check after acquiring lock (another coroutine may have refreshed)
        if (!tokenStorage.isTokenExpired()) {
            return@withLock tokenStorage.getToken()
        }

        val storedRefreshToken = tokenStorage.getRefreshToken()
            ?: run {
                tokenStorage.clearAuth()
                return@withLock null
            }

        val result = refreshToken(storedRefreshToken)
        result.fold(
            onSuccess = { response ->
                tokenStorage.saveGoTrueAuth(response)
                response.accessToken
            },
            onFailure = {
                Logger.w("PortalApiClient") { "Token refresh failed: ${it.message}" }
                tokenStorage.clearAuth()
                null
            }
        )
    }
}
```

**Update authenticatedRequest() to use ensureValidToken():**
```kotlin
private suspend inline fun <reified T> authenticatedRequest(
    block: (token: String) -> HttpResponse
): Result<T> {
    val token = ensureValidToken()
        ?: return Result.failure(PortalApiException(401, "Not authenticated - please log in again"))

    return try {
        val response = block(token)
        if (response.status.value == 401) {
            // Token was valid by our clock but server rejected it — force one refresh
            val retryToken = forceRefresh()
                ?: return Result.failure(PortalApiException(401, "Session expired - please log in again"))
            val retryResponse = block(retryToken)
            handleResponse(retryResponse)
        } else {
            handleResponse(response)
        }
    } catch (e: Exception) {
        Result.failure(e)
    }
}

/**
 * Force a token refresh regardless of local expiry state.
 * Used when server returns 401 despite local token appearing valid.
 */
private suspend fun forceRefresh(): String? {
    return refreshMutex.withLock {
        val storedRefreshToken = tokenStorage.getRefreshToken() ?: run {
            tokenStorage.clearAuth()
            return@withLock null
        }
        refreshToken(storedRefreshToken).fold(
            onSuccess = { response ->
                tokenStorage.saveGoTrueAuth(response)
                response.accessToken
            },
            onFailure = {
                tokenStorage.clearAuth()
                null
            }
        )
    }
}
```

```kotlin
/**
 * Restores a previous session on app startup.
 * If access_token exists but is expired, attempts a silent refresh.
 * Called from app init (VitruvianApp.onCreate or KoinInit).
 */
suspend fun restoreSession(): Result<Unit> {
    if (!tokenStorage.hasToken()) {
        // No stored session — user needs to log in
        return Result.success(Unit)
    }

    if (tokenStorage.isTokenExpired()) {
        // Access token expired — try refresh
        return refreshSession()
    }

    // Token exists and not expired — session is valid
    // Auth state is already derived from token storage flows in init{}
    return Result.success(Unit)
}
```

```kotlin
// Add constructor parameter
class PortalAuthRepository(
    private val apiClient: PortalApiClient,
    private val tokenStorage: PortalTokenStorage,
    private val userProfileRepository: UserProfileRepository  // NEW
) : AuthRepository {

override suspend fun signInWithEmail(email: String, password: String): Result<AuthUser> {
    return apiClient.signIn(email, password)
        .onSuccess { goTrueResponse ->
            tokenStorage.saveGoTrueAuth(goTrueResponse)
            linkUserProfile(goTrueResponse.user.id)
        }
        .map { goTrueResponse ->
            goTrueResponse.toPortalAuthResponse().user.toAuthUser()
        }
}

private suspend fun linkUserProfile(supabaseUserId: String) {
    try {
        // Link the active local profile to the Supabase user
        // NOTE: UserProfileRepository exposes activeProfile as StateFlow, not a suspend fun
        val activeProfile = userProfileRepository.activeProfile.value
        if (activeProfile != null && activeProfile.supabaseUserId != supabaseUserId) {
            userProfileRepository.linkToSupabase(activeProfile.id, supabaseUserId)
        }
    } catch (e: Exception) {
        Logger.w("PortalAuthRepository") { "Failed to link user profile: ${e.message}" }
        // Non-fatal — auth still succeeds
    }
}
```
